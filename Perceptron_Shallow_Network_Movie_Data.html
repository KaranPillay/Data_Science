def activate(weights, inputs):
    activation = weights[-1]
    for i in range(len(weights)-1):
        activation += weights[i] * inputs[i]
    return activation

def forward_propagate(network, row):
    inputs = row
    for i in range(len(network)):
        new_inputs = []
        for neuron in network[i]:
            activation = activate(neuron['w'], inputs)
            if i == len(network)-1:
                neuron['output'] = np.tanh(activation) #tanh
                new_inputs.append(neuron['output'])
            else:
                neuron['output'] = 1.0 / (1.0 + exp(-activation)) #linear
                new_inputs.append(neuron['output'])
        inputs = new_inputs
    return inputs

def backward_propagate(network, expected):
    for i in reversed(range(len(network))):
        layer = network[i]
        errors = list()
        if i != len(network)-1:
            for j in range(len(layer)):
                error = 0.0
                for neuron in network[i + 1]:
                    error += (neuron['w'][j] * neuron['delta'])
                errors.append(error)
        else:
            for j in range(len(layer)):
                neuron = layer[j]
                errors.append(expected[j] - neuron['output'])
                
        for j in range(len(layer)):
            neuron = layer[j]
            neuron['delta'] = errors[j] * (neuron['output'] * (1.0 - neuron['output'])) #derivative

def update_weights(network, row, l_rate):
    for i in range(len(network)):
        inputs = row[:-1]
        if i != 0:
            inputs = [neuron['output'] for neuron in network[i - 1]]
        for neuron in network[i]:
            for j in range(len(inputs)):
                neuron['w'][j] += l_rate * neuron['delta'] * inputs[j]
            neuron['w'][-1] += l_rate * neuron['delta']

def train_network( l_rate, n_epoch, meth):
    seed(123)
    it=[0,1] 
    m2["ans"]=0
    x=m2['x'].values
    y=m2['y'].values
    ou=m2['ans'].values

    train=[]
    for i in range(10):  
        train.append([x[i],y[i],1])

    n_inputs = len(train[0]) - 1
    n_outputs = len(train[0]) - 1
    network = list()
    hidden_layer = [{'w':[random() for i in range(n_inputs + 1)]} for i in range(1)]
    network.append(hidden_layer)
    output_layer = [{'w':[random() for i in range(1 + 1)]} for i in range(n_outputs)]
    network.append(output_layer)
    epo=[]
    er=[]
    net=[]
    outpu=[]
    for epoch in range(n_epoch):
        sum_error = 0
        for row in train:
            outputs = forward_propagate(network, row)
            expected = [0 for i in range(n_outputs)]
            expected[row[-1]] = 1
            sum_error += sum([(expected[i]-outputs[i])**2 for i in range(len(expected))])
            backward_propagate(network, expected)
            if(meth=="BGD"):
                update_weights(network, row, l_rate)
        
        if(meth=="SGD"):
                update_weights(network, row, l_rate)
                
        epo.append(epoch)
        er.append(sum_error)
        net.append(network)
        outpu.append(outputs)

    return epo,er,net,outpu
